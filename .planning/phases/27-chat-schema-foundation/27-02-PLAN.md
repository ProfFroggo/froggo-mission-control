After parsing `agentList` from `room.agents`, query the agents table to get names and roles:
```ts
if (agentList.length > 0) {
  // Look up agent names/roles for richer context
  const agentRows = agentList.map((aid: string) => {
    try {
      return db.prepare('SELECT id, name, role FROM agents WHERE id = ?').get(aid) as
        { id: string; name: string; role: string | null } | undefined;
    } catch { return undefined; }
  }).filter(Boolean);

  if (agentRows.length > 0) {
    const agentDesc = agentRows.map((a: { id: string; name: string; role: string | null }) =>
      a.role ? `${a.name} (${a.role})` : a.name
    ).join(', ');
    parts.push(`**Agents in room**: ${agentDesc}`);
  } else {
    parts.push(`**Agents**: ${agentList.join(', ')}`);
  }
}
```

This replaces the existing `parts.push(`**Agents**: ${agentList.join(', ')}`)` call with the richer version. Keep the fallback to raw IDs if DB lookup fails.
