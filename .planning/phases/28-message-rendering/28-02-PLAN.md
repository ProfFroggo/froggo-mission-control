```tsx
function formatDateSeparator(timestamp: number): string {
  const d = new Date(timestamp);
  const todayStart = new Date(); todayStart.setHours(0, 0, 0, 0);
  const yesterdayStart = new Date(todayStart); yesterdayStart.setDate(yesterdayStart.getDate() - 1);
  if (d >= todayStart) return 'Today';
  if (d >= yesterdayStart) return 'Yesterday';
  return d.toLocaleDateString([], { weekday: 'long', month: 'long', day: 'numeric' });
}

function isSameDay(ts1: number, ts2: number): boolean {
  const d1 = new Date(ts1); const d2 = new Date(ts2);
  return d1.getFullYear() === d2.getFullYear() &&
         d1.getMonth() === d2.getMonth() &&
         d1.getDate() === d2.getDate();
}

function DateSeparator({ label }: { label: string }) {
  return (
    <div className="flex items-center gap-3 my-4 px-4">
      <div className="flex-1 h-px bg-mission-control-border/50" />
      <span className="text-[11px] font-medium text-mission-control-text-dim/60 uppercase tracking-wide whitespace-nowrap">{label}</span>
      <div className="flex-1 h-px bg-mission-control-border/50" />
    </div>
  );
}
```

```tsx
function groupContentBlocks(blocks: ContentBlock[]): Array<ContentBlock | { type: '__group'; name: string; items: ContentBlock[] }> {
  const result: Array<ContentBlock | { type: '__group'; name: string; items: ContentBlock[] }> = [];
  let i = 0;
  while (i < blocks.length) {
    const b = blocks[i];
    if (b.type === 'tool_use' && b.name) {
      // Look ahead for consecutive same-tool blocks
      let j = i + 1;
      while (j < blocks.length && blocks[j].type === 'tool_use' && blocks[j].name === b.name) j++;
      if (j - i >= 3) {
        // Group 3+ consecutive identical tool calls
        result.push({ type: '__group', name: b.name, items: blocks.slice(i, j) });
        i = j;
        continue;
      }
    }
    result.push(b);
    i++;
  }
  return result;
}
```

Threshold is 3+ identical consecutive calls to avoid grouping legitimate single/pair calls.
