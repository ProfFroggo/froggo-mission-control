  Replace with secure version:
  ```typescript
  ipcMain.handle('ai:getAnalysis', async (_, id: string, platform: string) => {
    try {
      const row = prepare(
        "SELECT triage, summary, tasks, events, reply_draft, reply_needed FROM comms_ai_analysis WHERE external_id = ? AND platform = ?"
      ).get(id, platform) as any;

      if (!row) {
        return { success: true, analysis: null };
      }

      let tasks: any[] = [];
      let events: any[] = [];
      try { tasks = row.tasks ? JSON.parse(row.tasks) : []; } catch {}
      try { events = row.events ? JSON.parse(row.events) : []; } catch {}

      return {
        success: true,
        analysis: {
          triage: row.triage,
          summary: row.summary,
          tasks,
          events,
          reply_draft: row.reply_draft,
          reply_needed: !!row.reply_needed,
        },
      };
    } catch (e: any) {
      return { success: false, error: e.message };
    }
  });
  ```

  Key differences from backup:
  - Uses `prepare().get(id, platform)` with ? params instead of shell sqlite3 with string interpolation
  - No `runMsgCmd` or `exec` call — synchronous better-sqlite3
  - Guard JSON.parse on tasks/events fields with try/catch (the DB may contain malformed JSON)
  - No need for manual SQL escaping (`escapedId` pattern is gone)
