**2. Fix x_drafts CHECK constraint.**

The `x_drafts` table has `CHECK(status IN ('draft', 'approved', 'rejected', 'scheduled'))` but the `x:replyGuy:postNow` handler sets `status = 'posted'`. Add a migration to drop and recreate the constraint. In SQLite, you cannot alter CHECK constraints directly. Instead, add this migration:

```typescript
// Fix x_drafts CHECK constraint to allow 'posted' status
try {
  prepare(`CREATE TABLE IF NOT EXISTS x_drafts_new (
    id TEXT PRIMARY KEY,
    plan_id TEXT,
    version TEXT NOT NULL,
    content TEXT NOT NULL,
    media_paths TEXT,
    proposed_by TEXT NOT NULL,
    approved_by TEXT,
    status TEXT NOT NULL DEFAULT 'draft',
    created_at INTEGER NOT NULL,
    updated_at INTEGER,
    file_path TEXT,
    metadata TEXT,
    FOREIGN KEY(plan_id) REFERENCES x_content_plans(id),
    CHECK(status IN ('draft', 'approved', 'rejected', 'scheduled', 'posted'))
  )`).run();

  // Only migrate if the new table was just created (empty) and old table exists
  const oldCount = (prepare('SELECT COUNT(*) as cnt FROM x_drafts').get() as any)?.cnt ?? 0;
  const newCount = (prepare('SELECT COUNT(*) as cnt FROM x_drafts_new').get() as any)?.cnt ?? 0;

  if (newCount === 0 && oldCount >= 0) {
    prepare('INSERT INTO x_drafts_new SELECT * FROM x_drafts').run();
    prepare('DROP TABLE x_drafts').run();
    prepare('ALTER TABLE x_drafts_new RENAME TO x_drafts').run();
    // Recreate indexes
    prepare('CREATE INDEX IF NOT EXISTS idx_drafts_status ON x_drafts(status)').run();
    prepare('CREATE INDEX IF NOT EXISTS idx_drafts_plan ON x_drafts(plan_id)').run();
    prepare('CREATE INDEX IF NOT EXISTS idx_drafts_version ON x_drafts(plan_id, version)').run();
    prepare('CREATE INDEX IF NOT EXISTS idx_drafts_created ON x_drafts(created_at DESC)').run();
  }
} catch (e: any) {
  // Already migrated or table doesn't exist yet
  safeLog.log('[Migration] x_drafts CHECK constraint migration:', e.message);
}
```
