**1. Add a useEffect that listens for `x-agent-chat-inject` events:**

Inside the `XAgentChatPane` component (after the existing useEffect blocks, around line 97), add:

```typescript
// Listen for external message injection (e.g., "Suggest Reply" from XReplyGuyView)
useEffect(() => {
  const handler = (e: Event) => {
    const message = (e as CustomEvent).detail?.message as string;
    if (message && !loading) {
      setInput(message);
      // Auto-send after a brief delay so the input renders first
      setTimeout(() => {
        const sendBtn = document.querySelector('[data-agent-chat-send]') as HTMLButtonElement;
        sendBtn?.click();
      }, 100);
    }
  };
  window.addEventListener('x-agent-chat-inject', handler);
  return () => window.removeEventListener('x-agent-chat-inject', handler);
}, [loading]);
```
