**feedbackStore.ts** — Zustand store at `src/store/feedbackStore.ts`:

```typescript
import { create } from 'zustand';

interface FeedbackState {
  selectedAgent: string;       // 'writer' | 'researcher' | 'jess'
  instructions: string;
  streaming: boolean;
  streamContent: string;       // raw accumulated stream text
  alternatives: string[];      // parsed alternatives (set on stream end)
  error: string | null;
  // Selection range saved at send time (for accept to use after stream)
  savedSelection: { from: number; to: number } | null;

  setSelectedAgent: (agent: string) => void;
  setInstructions: (text: string) => void;
  setStreaming: (streaming: boolean) => void;
  setStreamContent: (content: string) => void;
  setAlternatives: (alts: string[]) => void;
  setError: (error: string | null) => void;
  setSavedSelection: (sel: { from: number; to: number } | null) => void;
  reset: () => void;
}

export const useFeedbackStore = create<FeedbackState>((set) => ({
  selectedAgent: 'writer',
  instructions: '',
  streaming: false,
  streamContent: '',
  alternatives: [],
  error: null,
  savedSelection: null,

  setSelectedAgent: (agent) => set({ selectedAgent: agent }),
  setInstructions: (text) => set({ instructions: text }),
  setStreaming: (streaming) => set({ streaming }),
  setStreamContent: (content) => set({ streamContent: content }),
  setAlternatives: (alts) => set({ alternatives: alts }),
  setError: (error) => set({ error }),
  setSavedSelection: (sel) => set({ savedSelection: sel }),
  reset: () => set({
    instructions: '',
    streaming: false,
    streamContent: '',
    alternatives: [],
    error: null,
    savedSelection: null,
  }),
}));
```

Key design notes:
- `savedSelection` stores the editor range at send time so accept can replace even if selection changed during streaming
- `reset()` does NOT clear `selectedAgent` — user's agent preference persists across interactions
- No bridge() calls — this is pure client state, not IPC-backed

**writing-feedback-service.ts** — Electron IPC service at `electron/writing-feedback-service.ts`:

Follow the exact pattern from `writing-project-service.ts` (registerXxxHandlers function, ipcMain.handle).

```typescript
import { ipcMain } from 'electron';
import * as fs from 'fs';
import * as path from 'path';
import { writingMemoryPath } from './paths';

interface FeedbackLogEntry {
  timestamp: string;
  chapterId: string;
  agentId: string;
  selectedText: string;
  instructions: string;
  alternatives: string[];
  accepted: string | null;  // null if dismissed
  selectionRange: { from: number; to: number };
}

export function registerWritingFeedbackHandlers() {
  // Log a feedback interaction (append JSONL)
  ipcMain.handle('writing:feedback:log', async (_event, projectId: string, entry: FeedbackLogEntry) => {
    try {
      const memoryDir = writingMemoryPath(projectId, '');
      await fs.promises.mkdir(path.dirname(memoryDir), { recursive: true });
      // Ensure memory directory exists
      const dirPath = path.join(path.dirname(writingMemoryPath(projectId, 'x')), '');
      // writingMemoryPath gives {projectId}/memory/{filename}
      // We need to ensure the memory/ directory exists
      const logPath = writingMemoryPath(projectId, `feedback-${entry.chapterId}.jsonl`);
      await fs.promises.mkdir(path.dirname(logPath), { recursive: true });
      const line = JSON.stringify({ ...entry, timestamp: new Date().toISOString() }) + '\n';
      await fs.promises.appendFile(logPath, line, 'utf-8');
      return { success: true };
    } catch (err: any) {
      console.error('[writing-feedback] log error:', err);
      return { success: false, error: err.message };
    }
  });

  // Read feedback history for a chapter
  ipcMain.handle('writing:feedback:history', async (_event, projectId: string, chapterId: string) => {
    try {
      const logPath = writingMemoryPath(projectId, `feedback-${chapterId}.jsonl`);
      const exists = await fs.promises.access(logPath).then(() => true).catch(() => false);
      if (!exists) return { success: true, entries: [] };
      const raw = await fs.promises.readFile(logPath, 'utf-8');
      const entries = raw.trim().split('\n').filter(Boolean).map(line => JSON.parse(line));
      return { success: true, entries };
    } catch (err: any) {
      console.error('[writing-feedback] history error:', err);
      return { success: false, error: err.message, entries: [] };
    }
  });
}
```

Simplify the mkdir logic — `writingMemoryPath(projectId, filename)` returns `~/froggo/writing-projects/{projectId}/memory/{filename}`. Use `path.dirname(logPath)` to get the memory directory and mkdir with recursive.
