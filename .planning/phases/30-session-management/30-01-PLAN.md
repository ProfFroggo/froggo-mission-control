**Step 2: Replace the delete handler**

Find the existing delete handler (around line 1047-1054 where `chatApi.deleteSession` is called). Replace with:

```tsx
const handleDeleteSession = useCallback(async (agentId: string, sessionKey: string) => {
  // Cancel any existing pending delete
  if (pendingDeleteRef.current) {
    clearTimeout(pendingDeleteRef.current.timer);
    // Commit the previous delete immediately before starting new one
    const prev = pendingDeleteRef.current;
    await chatApi.deleteSession(prev.agentId); // commit prior
  }

  setPendingDeleteId(agentId);

  const timer = setTimeout(async () => {
    // Commit delete after 5 seconds
    try {
      await chatApi.deleteSession(sessionKey);
      // Remove agent from local cache/state
      messageCacheRef.current.delete(agentId);
    } catch (err) {
      console.error('Session delete failed:', err);
      showToast('Failed to delete session', 'error');
    } finally {
      setPendingDeleteId(null);
      pendingDeleteRef.current = null;
    }
  }, 5000);

  pendingDeleteRef.current = { agentId, timer };

  // Show undo toast
  showToast(
    `Session deleted — click to undo`,
    'info',
    // If showToast supports actions, add undo callback:
    // { action: 'Undo', onAction: () => handleUndoDelete() }
  );
}, []);

const handleUndoDelete = useCallback(() => {
  if (!pendingDeleteRef.current) return;
  clearTimeout(pendingDeleteRef.current.timer);
  pendingDeleteRef.current = null;
  setPendingDeleteId(null);
  showToast('Delete cancelled', 'info');
}, []);
```

**Note about showToast**: Read the Toast component first to check if it supports an action button. If not, the undo pattern uses a separate small banner instead.
