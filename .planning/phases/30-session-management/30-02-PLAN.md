**Step 2: Derive per-agent status from room messages**

Add a `useEffect` or `useMemo` to compute per-agent status:

```tsx
const agentStatuses = useMemo((): Record<string, AgentStatus> => {
  const statuses: Record<string, AgentStatus> = {};
  const messages = room?.messages || [];
  const now = Date.now();

  // Check each room agent's last message
  const roomAgentIds: string[] = (() => {
    try { return JSON.parse(room?.agents || '[]'); } catch { return []; }
  })();

  for (const agentId of roomAgentIds) {
    // Find last message from this agent (most recent first)
    const lastMsg = [...messages].reverse().find(m => m.agentId === agentId);

    if (!lastMsg) {
      statuses[agentId] = 'idle';
      continue;
    }

    // Currently streaming = thinking
    if (lastMsg.streaming) {
      statuses[agentId] = 'thinking';
      continue;
    }

    // Responded within last 30 seconds = done (brief green indicator)
    if (lastMsg.timestamp && now - lastMsg.timestamp < 30_000) {
      statuses[agentId] = 'done';
      continue;
    }

    statuses[agentId] = 'idle';
  }

  return statuses;
}, [room?.messages, room?.agents]);
```
