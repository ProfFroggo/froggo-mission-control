    **1. Create `src/store/writingStore.ts`:**

    A Zustand store (NOT persisted — no localStorage for document content, per pitfall M1):

    ```typescript
    import { create } from 'zustand';

    export interface WritingProject {
      id: string;
      title: string;
      type: 'memoir' | 'novel';
      chapterCount: number;
      wordCount: number;
      createdAt: number;
      updatedAt: number;
    }

    export interface WritingChapter {
      id: string;
      title: string;
      filename: string;
      position: number;
      wordCount: number;
      createdAt: number;
      updatedAt: number;
    }

    interface WritingState {
      // Project list
      projects: WritingProject[];
      projectsLoading: boolean;

      // Active project
      activeProjectId: string | null;
      activeProject: (WritingProject & { chapters: WritingChapter[] }) | null;

      // Active chapter
      activeChapterId: string | null;
      activeChapterContent: string | null;
      chapterLoading: boolean;
      chapterDirty: boolean;  // unsaved changes

      // Actions
      loadProjects: () => Promise<void>;
      createProject: (title: string, type: 'memoir' | 'novel') => Promise<string | null>;
      deleteProject: (projectId: string) => Promise<void>;
      openProject: (projectId: string) => Promise<void>;
      closeProject: () => void;

      openChapter: (chapterId: string) => Promise<void>;
      closeChapter: () => void;
      createChapter: (title: string) => Promise<string | null>;
      renameChapter: (chapterId: string, title: string) => Promise<void>;
      deleteChapter: (chapterId: string) => Promise<void>;
      reorderChapters: (chapterIds: string[]) => Promise<void>;

      saveChapter: (content: string) => Promise<void>;
      setChapterDirty: (dirty: boolean) => void;
      setActiveChapterContent: (content: string) => void;
    }
    ```

    Implement each action using `window.clawdbot.writing.*` bridge calls. Cast window as `any` for clawdbot access (matches existing pattern across the codebase).

    Key implementation details:
    - `loadProjects()`: calls `window.clawdbot.writing.project.list()`, sets `projects` and `projectsLoading`
    - `createProject()`: calls bridge, then calls `loadProjects()` to refresh list, returns new project ID
    - `openProject()`: calls `window.clawdbot.writing.project.get(id)`, sets `activeProjectId` and `activeProject` with chapters
    - `openChapter()`: calls `window.clawdbot.writing.chapter.read(projectId, chapterId)`, sets `activeChapterId`, `activeChapterContent`, clears `chapterDirty`
    - `saveChapter()`: calls bridge, clears `chapterDirty`, refreshes activeProject to update word counts
    - `closeProject()`: resets `activeProjectId`, `activeProject`, `activeChapterId`, `activeChapterContent`
