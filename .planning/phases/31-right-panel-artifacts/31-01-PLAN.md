**Step 1: Add SessionContextPanel component**

```tsx
interface SessionContextPanelProps {
  messages: StructuredChatMessage[];
  isOpen: boolean;
  onClose: () => void;
}

function SessionContextPanel({ messages, isOpen, onClose }: SessionContextPanelProps) {
  const [activeTab, setActiveTab] = useState<'progress' | 'files'>('progress');

  // Extract tool calls from all messages
  const toolCalls = useMemo(() => {
    const calls: Array<{ name: string; input: any; timestamp: number; msgId: string }> = [];
    for (const msg of messages) {
      if (!Array.isArray(msg.content)) continue;
      for (const block of msg.content) {
        if (block.type === 'tool_use' && block.name) {
          calls.push({
            name: block.name,
            input: block.input,
            timestamp: msg.timestamp ?? 0,
            msgId: msg.id,
          });
        }
      }
    }
    return calls;
  }, [messages]);

  // Extract file paths from tool inputs
  const workingFiles = useMemo(() => {
    const files = new Set<string>();
    for (const call of toolCalls) {
      const inp = call.input || {};
      // Common file path keys across Claude tools
      if (typeof inp.file_path === 'string') files.add(inp.file_path);
      if (typeof inp.path === 'string') files.add(inp.path);
      if (typeof inp.new_path === 'string') files.add(inp.new_path);
      // Extract file paths from bash commands (heuristic)
      if (typeof inp.command === 'string') {
        const pathMatches = inp.command.match(/(?:^|\s)(\/[\w./\-_]+\.(?:ts|tsx|js|jsx|py|json|md|css|html))/g);
        if (pathMatches) pathMatches.forEach(p => files.add(p.trim()));
      }
    }
    return [...files].filter(f => f.length > 3);
  }, [toolCalls]);

  return (
    <div className={`flex flex-col border-l border-mission-control-border bg-mission-control-surface transition-all duration-200 ${
      isOpen ? 'w-72 min-w-72' : 'w-0 overflow-hidden'
    }`}>
      {isOpen && (
        <>
          {/* Header */}
          <div className="flex items-center justify-between px-3 py-2.5 border-b border-mission-control-border">
            <div className="flex gap-1">
              {(['progress', 'files'] as const).map(tab => (
                <button key={tab} onClick={() => setActiveTab(tab)}
                  className={`px-3 py-1 text-xs rounded-md font-medium transition-colors ${
                    activeTab === tab
                      ? 'bg-mission-control-accent/15 text-mission-control-accent'
                      : 'text-mission-control-text-dim hover:text-mission-control-text'
                  }`}>
                  {tab === 'progress' ? 'Progress' : `Files${workingFiles.length > 0 ? ` (${workingFiles.length})` : ''}`}
                </button>
              ))}
            </div>
            <button onClick={onClose} className="p-1 rounded hover:bg-mission-control-border/50 text-mission-control-text-dim hover:text-mission-control-text">
              <X size={14} />
            </button>
          </div>

          {/* Progress Tab */}
          {activeTab === 'progress' && (
            <div className="flex-1 overflow-y-auto p-3 space-y-2">
              {toolCalls.length === 0 ? (
                <p className="text-xs text-mission-control-text-dim/60 text-center py-8">No tool activity yet</p>
              ) : (
                toolCalls.map((call, i) => (
                  <div key={i} className="flex items-start gap-2">
                    <div className="mt-1 w-1.5 h-1.5 rounded-full bg-mission-control-accent/50 flex-shrink-0" />
                    <div className="flex-1 min-w-0">
                      <p className="text-xs font-medium text-mission-control-text truncate">
                        {call.name.replace(/_/g, ' ')}
                      </p>
                      {call.input?.file_path && (
                        <p className="text-[10px] text-mission-control-text-dim/60 truncate font-mono">
                          {call.input.file_path.split('/').slice(-2).join('/')}
                        </p>
                      )}
                    </div>
                  </div>
                ))
              )}
            </div>
          )}

          {/* Files Tab */}
          {activeTab === 'files' && (
            <div className="flex-1 overflow-y-auto p-3 space-y-1.5">
              {workingFiles.length === 0 ? (
                <p className="text-xs text-mission-control-text-dim/60 text-center py-8">No files touched yet</p>
              ) : (
                workingFiles.map((file, i) => (
                  <div key={i} className="flex items-center gap-2 px-2 py-1.5 rounded hover:bg-mission-control-bg/50 group">
                    <FileCode size={12} className="text-mission-control-text-dim/50 flex-shrink-0" />
                    <span className="text-xs text-mission-control-text font-mono truncate" title={file}>
                      {file.split('/').slice(-2).join('/')}
                    </span>
                  </div>
                ))
              )}
            </div>
          )}
        </>
      )}
    </div>
  );
}
```
